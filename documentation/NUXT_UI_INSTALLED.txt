💡 QUICK EXAMPLES:

   Button:
   <UButton>Click Me</UButton>
   <UButton color="primary" icon="i-heroicons-rocket-launch">Launch</UButton>

   Card:
   <UCard>
     <template #header>Title</template>
     Content
   </UCard>

   Alert:
   <UAlert title="Success!" color="success" />

   Form:
   <UFormGroup label="Email">
     <UInput v-model="email" type="email" />
   </UFormGroup>

   Modal:
   <UButton @click="isOpen = true">Open</UButton>
   <UModal v-model="isOpen">
     <UCard>Content</UCard>
   </UModal>
